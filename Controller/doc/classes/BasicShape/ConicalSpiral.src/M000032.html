                def create (par={})
                        @vr = par[:vr]
                        @vz = par[:vz]
                        @omega = par[:omega]
                        @ri = par[:ri]
                        @zi = par[:zi]
                        @rtm = par[:rtm]
                        p = Array.new(3)
                        proc do |t|
                                p1 = Matrix.column_vector [(@ri+@vr*t)*Math::cos(@omega*t),
                                                                                  (@ri+@vr*t)*Math::sin(@omega*t),
                                                                                        @zi+@vz*t,
                                                                                       1.0]
                                p1 = @rtm * p1
                                p1.column(0).to_a[0..2]
                        end                  
                end
