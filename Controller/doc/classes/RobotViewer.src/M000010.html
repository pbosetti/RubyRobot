  def special
    special = lambda do |key,x,y|
      case key
      when GLUT_KEY_HOME
        @x_rot = 45
        @y_rot = -135
        @z_rot = 0
        @x_incr = 0
        @y_incr = 0
        @scale_factor = 1.0
      when GLUT_KEY_LEFT
        @y_rot -= 10
      when GLUT_KEY_RIGHT
        @y_rot += 10
      when GLUT_KEY_UP
        @scale_factor *= 1.1
      when GLUT_KEY_DOWN
        @scale_factor /= 1.1
      end
    end
  end
