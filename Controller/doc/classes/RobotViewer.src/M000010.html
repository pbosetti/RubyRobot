  def mouse
    lambda do |button, state, x, y|
      @modifiers = glutGetModifiers()
      if (button == GLUT_LEFT_BUTTON)
        if (state == GLUT_UP)
          @x_last = -1
          @y_last = -1
        end
        @x_incr = 0
        @y_incr = 0
      end
    end
  end
