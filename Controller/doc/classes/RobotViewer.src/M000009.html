  def keyboard
    lambda do |key, x, y|
      case key
      when ?a
        @draw[:axes] = !@draw[:axes]
      when ?\e
        exit(0)
      when ?w
        @draw[:wireframe] = !@draw[:wireframe]
      else
        puts "Keystrokes:\na           toggles axes visibility\nw           toggle wireframe\nUp/Down     rotate about Y (green)\nLeft/Right  rotate about Z (blue)\nHome        reset rotation and scale\nESC         exit\n"
      end
    end
  end
